---
title: "List comments"
api: "GET /api/v1/projects/{projectId}/issues/{issueId}/comments"
description: "Returns all comments for an issue."
---

## Authentication

<Icon icon="lock" /> Requires `Authorization: Bearer <token>`

## Path parameters

<ParamField path="projectId" type="string" required>
  UUID of the project.
</ParamField>

<ParamField path="issueId" type="string" required>
  UUID of the issue.
</ParamField>

## Response

<ResponseField name="data" type="array">
  Array of comment objects.

  <Expandable title="comment object">
    <ResponseField name="id" type="string">Comment UUID</ResponseField>
    <ResponseField name="issueId" type="string">Parent issue UUID</ResponseField>
    <ResponseField name="parentCommentId" type="string | null">Parent comment UUID for replies</ResponseField>
    <ResponseField name="linearCommentId" type="string">Linked Linear comment ID</ResponseField>
    <ResponseField name="body" type="string">Comment text</ResponseField>
    <ResponseField name="authorName" type="string | null">Author display name</ResponseField>
    <ResponseField name="authorId" type="string | null">Author identifier</ResponseField>
    <ResponseField name="edited" type="boolean">Whether the comment was edited</ResponseField>
    <ResponseField name="createdAt" type="string">ISO 8601 timestamp</ResponseField>
    <ResponseField name="updatedAt" type="string">ISO 8601 timestamp</ResponseField>
  </Expandable>
</ResponseField>
