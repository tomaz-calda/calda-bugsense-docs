---
title: "List issues"
api: "GET /api/v1/projects/{projectId}/issues"
description: "Returns all issues for a project."
---

## Authentication

<Icon icon="lock" /> Requires `Authorization: Bearer <token>`

## Path parameters

<ParamField path="projectId" type="string" required>
  UUID of the project.
</ParamField>

## Response

<ResponseField name="data" type="array">
  Array of issue objects.

  <Expandable title="issue object">
    <ResponseField name="id" type="string">Issue UUID</ResponseField>
    <ResponseField name="title" type="string">Bug title</ResponseField>
    <ResponseField name="description" type="string | null">Bug description</ResponseField>
    <ResponseField name="platform" type="string | null">`web`, `testflight`, `android`, or `figma`</ResponseField>
    <ResponseField name="environment" type="string | null">`development`, `staging`, or `production`</ResponseField>
    <ResponseField name="appVersion" type="string | null">Application version</ResponseField>
    <ResponseField name="projectId" type="string">Parent project UUID</ResponseField>
    <ResponseField name="linearTaskId" type="string | null">Linked Linear issue ID</ResponseField>
    <ResponseField name="linearStatus" type="string | null">Current status from Linear</ResponseField>
    <ResponseField name="linearPriority" type="integer | null">Priority level (0–4)</ResponseField>
    <ResponseField name="linearAssigneeName" type="string | null">Assignee name from Linear</ResponseField>
    <ResponseField name="linearUrl" type="string | null">Direct link to Linear issue</ResponseField>
    <ResponseField name="linearTicketNumber" type="string | null">Ticket ID (e.g. `BUG-42`)</ResponseField>
    <ResponseField name="linearLabels" type="string[] | null">Labels from Linear</ResponseField>
    <ResponseField name="archivedAt" type="string | null">ISO 8601 timestamp if archived</ResponseField>
    <ResponseField name="createdBy" type="string | null">UUID of the user who created the issue</ResponseField>
    <ResponseField name="createdAt" type="string">ISO 8601 timestamp</ResponseField>
    <ResponseField name="updatedAt" type="string">ISO 8601 timestamp</ResponseField>
  </Expandable>
</ResponseField>
